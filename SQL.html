-- 1. Tabla de Usuarios CREATE TABLE Usuarios ( id_usuario INT PRIMARY KEY AUTO_INCREMENT, nombre VARCHAR(50) ); -- 2. Tabla para Bienestar Diario CREATE TABLE Registro_Diario ( id_registro INT PRIMARY KEY AUTO_INCREMENT, id_usuario INT, fecha DATE, horas_sueno DECIMAL(4,2), nivel_estres INT CHECK (nivel_estres BETWEEN 1 AND 10), FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) ); -- 3. Tabla para Hobbies (Lectura y Caminata) CREATE TABLE Hobbie_Actividad ( id_actividad INT PRIMARY KEY AUTO_INCREMENT, id_usuario INT, fecha DATE, tipo_actividad VARCHAR(50), -- Ej: "Lectura", "Caminata" duracion_minutos INT, -- Ej: 45 progreso_logrado DECIMAL(5,2), -- Ej: 20 (páginas) o 3.5 (km) detalle_extra VARCHAR(100), -- Ej: "Crimen y Castigo" o "Parque central" FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) ); -- 4. Tabla de Finanzas CREATE TABLE Finanzas ( id_transaccion INT PRIMARY KEY AUTO_INCREMENT, id_usuario INT, fecha DATE, tipo VARCHAR(20), concepto VARCHAR(100), monto DECIMAL(10,2), FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) );
